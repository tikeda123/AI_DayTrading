import os
import sys
import json
from dependency_injector import containers, providers



from common.config_manager import ConfigManager
from common.trading_logger import TradingLogger
from common.data_loader import DataLoader
from common.data_loader_db import DataLoaderDB


# 現在のスクリプトのディレクトリの絶対パスを取得
current_script_path = os.path.dirname(os.path.abspath(__file__))
# 設定ファイルが存在するディレクトリのパスを構築（親ディレクトリ）
config_file_directory = os.path.join(current_script_path, os.pardir)
# 設定ファイルの相対パスを構築
config_path = os.path.join(config_file_directory, 'aitrading_settings_ver2.json')
# 相対パスを絶対パスに変換
config_path = os.path.abspath(config_path)

class Container(containers.DeclarativeContainer):
    """
    アプリケーションの依存関係と設定を管理するコンテナ。

    このコンテナは、設定ファイルから設定を読み込み、アプリケーション全体で使用される
    コンポーネントのインスタンスを提供します。

    Attributes:
        config_path (str): 設定ファイルへのパス。
        config (providers.Configuration): 設定情報を保持するプロバイダ。
        config_manager (providers.Factory): ConfigManagerのファクトリプロバイダ。
        trading_logger (providers.Singleton): TradingLoggerのシングルトンプロバイダ。
        trading_logger_singleton (providers.Singleton): TradingLoggerインスタンスのシングルトンプロバイダ。
        data_loader (providers.Factory): DataLoaderのファクトリプロバイダ。
        data_loader_db (providers.Factory): DataLoaderDBのファクトリプロバイダ。
    """
    # 環境変数から設定ファイルのパスを取得、または上で計算したパスを使用
    config_path = os.environ.get('CONFIG_PATH', config_path)

    # 設定ファイルを開いてJSONとして読み込む
    with open(config_path, 'r') as config_file:
        config_data = json.load(config_file)
    config = providers.Configuration()
    config.update(config_data)

    config_manager = providers.Factory(
        ConfigManager,
        config_fullpath=config_path
    )

    trading_logger = providers.Singleton(
        TradingLogger,
        conf=config.LOG
    )

    trading_logger_singleton = providers.Singleton(
        TradingLogger.get_instance
    )

    data_loader = providers.Factory(
        DataLoader,
        conf=config.DATA
    )

    data_loader_db = providers.Factory(
        DataLoaderDB,
        config_fullpath=config_path
    )

def main():
    # 環境変数CONFIG_PATHを必要に応じて設定
    # os.environ['CONFIG_PATH'] = config_path  # 必要な場合、コメントを外す
    container = Container()
    container.wire(modules=[__name__])

    mg = container.config_manager()
    lg = container.trading_logger()
    lg.log_system_message('test')

    dl = container.data_loader()

    for i in range(10):
        dl = container.data_loader()

    print(dl.get_raw())

if __name__ == '__main__':
    main()
